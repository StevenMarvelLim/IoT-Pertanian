Create new database with the name 'iot_pertanian', then create the table using this code below:


CREATE TABLE sensor_data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    timestamp DATETIME NOT NULL,
    temperature FLOAT,
    humidity FLOAT,
    ldrValue FLOAT,
    rainValue FLOAT,
    airQualityPPM FLOAT,
    soilMoisture FLOAT
);
